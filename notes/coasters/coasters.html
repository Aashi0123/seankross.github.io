The Riddler

A quick visualization of Eric’s readings from his accelerometer.

# Create temporary file
temp <- tempfile()

# Download data from GitHub
download.file("https://raw.githubusercontent.com/weissometer/6-Flags-Accelerometer-Data/master/Readings%20from%206.12.14/the%20riddler_acc.csv", temp, method = "curl")

# Read data into data frame
acc <- read.csv(temp)

# Plot values,
# then rinse and repeat for other roller coasters
plot(acc$Z_value, type="l", ylim=c(-20, 15), xlab="Time", ylab="Data", main="The Riddler")
lines(acc$X_value, col="red")
lines(acc$Y_value, col="blue")
legend(0, -10, c("X", "Y", "Z"), lty=c(1,1,1), lwd=c(2, 2, 2), col=c("red", "blue", "black"))
